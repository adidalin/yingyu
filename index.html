function render() { const content = this.content; // Define fixed font sizes const wordFontSize = '14px'; // Fixed size for words const sentenceFontSize = '16px'; // Fixed size for sentences // Clear previous content this.container.innerHTML = ''; // Create elements for rendering const elements = content.split(/(?<=\.|!|\?)\s+/).map(sentence => { const sentenceElement = document.createElement('div'); sentenceElement.style.fontSize = sentenceFontSize; sentenceElement.textContent = sentence; // Split sentence into words const words = sentence.split(' '); words.forEach(word => { const wordElement = document.createElement('span'); wordElement.style.fontSize = wordFontSize; wordElement.textContent = word + ' '; sentenceElement.appendChild(wordElement); }); return sentenceElement; }); // Append new content to container elements.forEach(element => this.container.appendChild(element)); }
